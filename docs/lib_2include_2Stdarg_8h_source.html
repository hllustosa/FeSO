70 /*
71  * Amount of space required in an argument list (ie. the stack) for an
72  * argument of type t.
73  */
74 #define __va_argsiz(t) \
75  (((sizeof(t) + sizeof(int) - 1) / sizeof(int)) * sizeof(int))
76 
77 
78 /*
79  * Start variable argument list processing by setting AP to point to the
80  * argument after pN.
81  */
82 #ifdef __GNUC__
83 /*
84  * In GNU the stack is not necessarily arranged very neatly in order to
85  * pack shorts and such into a smaller argument list. Fortunately a
86  * neatly arranged version is available through the use of __builtin_next_arg.
87  */
88 #define va_start(ap, pN) \
89  ((ap) = ((va_list) __builtin_next_arg(pN)))
90 #else
91 /*
92  * For a simple minded compiler this should work (it works in GNU too for
93  * vararg lists that don't follow shorts and such).
94  */
95 #define va_start(ap, pN) \
96  ((ap) = ((va_list) (&pN) + __va_argsiz(pN)))
97 #endif
98 
99 
100 /*
101  * End processing of variable argument list. In this case we do nothing.
102  */
103 #define va_end(ap) ((void)0)
104 
105 
106 /*
107  * Increment ap to the next argument in the list while returing a
108  * pointer to what ap pointed to first, which is of type t.
109  *
110  * We cast to void* and then to t* because this avoids a warning about
111  * increasing the alignment requirement.
112  */
113 
114 #define va_arg(ap, t) \
115  (((ap) = (ap) + __va_argsiz(t)), \
116  *((t*) (void*) ((ap) - __va_argsiz(t))))
117 
118 #endif /* Not RC_INVOKED */
